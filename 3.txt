#include <iostream>
#include <vector>
#include <algorithm>
#include <math.h>
using namespace std;

int main() {
 long long int n;
 vector <long long int> v;
 cin>>n;
 long long int p;
 
 while(n%2==0)
 {
 	v.push_back(2);
 	cout<<2<<endl;
 	n = n / 2;
 }

for(long int i=3;i<=n;i=i+2)
{
//	cout<<"SQRT(N)="<<sqrt(n)<<endl;
	while(n%i==0)
	{
		v.push_back(i);
		//cout<<i<<endl;
		n = n / i;
	}
	
//	cout<<"I= "<<i<<endl;
}

if(v.empty())
{
	cout<<n; // the number itself is prime 
}
else
{
	p = *max_element(v.begin(),v.end());
	cout<<p;
}

}
